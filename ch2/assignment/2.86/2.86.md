[s][e][x][frac]
[1][k][1][n]
k = 15, n = 63
bias = 2^(k-1) - 1 = 2^14 - 1

                          值                    十进制
最小的正非规格化数   [0][00000][0][000...1]    2^(1-bias-63)
    [0][00000][0][000...1]
    f = 0.0000..1
    M = 0.0000..1 = 2^(-n)
    bias = 2^(k-1) - 1
    E = 1 - bias
    V = 2^E * M = 2^(1-bias-n)

最小的正规格化数    [0][00..1][1][0000]         2^(1-bias)
    [0][00..1][1][0000]
    M = 1.0 = 1
    f = 0.0000
    e = 1
    E = e - bias = 1 - bias
    V = 2^(1-bias)

最大的规格化数     [0][111...0][1][11111]    2^bias * (2 - 2^(-63))
    [0][111...0][1][11111]
    f = 0.1111 = 1 - 2^(-n)
    M = 1 + f = 2 - 2^(-n)
    e = 2^k - 1 - 1 (不能全1，所以再减1)
    E = e - bias = 2^k - 2 - (2^(k-1) - 1)
      = 2^(k-1) - 1 = bias
    V = 2^E * M = 2^bias * (2 - 2^(-n))
    V = 2^bias * (2 - 2^(-63))
